fun sumOfN: int v[100], int n {
	int sum is 0;
	for: int i from 0 to n-1
		/ sum is sum+v[i];
	ret sum;
} ret int;

 
fun maxOfN: int v[100], int n {
	int max is -10000;
	for: int i from 0 to n-1
		/ if: v[i] > max
			/ max is v[i];
	ret max;
} ret int;


fun minOfN: int v[100], int n {
	int min is 10000;
	for: int i from 0 to n-1
		/ if: v[i] < min
			/ min is v[i];
	ret min;
} ret int;
